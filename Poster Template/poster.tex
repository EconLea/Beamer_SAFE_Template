\newcommand{\numcolumns}{2} 

% Calculate widths dynamically
\newlength{\colwidth}

% (N+1)*sep + N*col = \paperwidth
% Let sepwidth = s => colwidth = c = (\paperwidth - (N+1)*s)/N
% Set sepwidth as fraction of paperwidth (e.g., 0.02\paperwidth)
% \setlength{\sepwidth}{0.05\paperwidth}

% Compute colwidth = (\paperwidth - (\numcolumns + 1)*\sepwidth)/\numcolumns
\pgfmathsetmacro{\calccolwidth}{(\paperwidth - (\numcolumns + 1)*\sepwidth) / \numcolumns}
\setlength{\colwidth}{\calccolwidth pt}
