% Define truly “safe” lists that always use a colored textbullet / label
%─────────────────────────────────────────────────────────────────────────────
\newlist{safeitemize}{itemize}{3}
\setlist[safeitemize,1]{label=\color{primary}\large\textbullet, leftmargin=*}
\setlist[safeitemize,2]{label=\color{primary}\normalsize\textbullet, leftmargin=1.5em}
\setlist[safeitemize,3]{label=\color{primary}\scriptsize\textbullet, leftmargin=3em}

\newlist{safenumerate}{enumerate}{3}
\setlist[safenumerate,1]{label=\color{primary}\large\arabic*., leftmargin=*}
\setlist[safenumerate,2]{label=\color{primary}\normalsize\arabic*., leftmargin=1.5em}
\setlist[safenumerate,3]{label=\color{primary}\scriptsize\arabic*., leftmargin=3em}
